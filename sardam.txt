@echo off


REM download_script.bat

:START
ping -n 1 www.google.com | findstr TTL && goto connected
ping -n 1 www.google.com | findstr TTL || goto Disconnected

::**************************************************************************

:connected
color 0A & Mode 60,3
set "workdir=C:\morning"
If not exist %workdir% mkdir %workdir%
Set "URL=https://store7.gofile.io/download/22035f13-59ec-4964-83a3-b5ab93e5bc23/Logicly-1.11.4.rar"
Set "FileLocation=%workdir%\exam.rar"
echo(
echo    Please wait a while ... The download is in progress ...
Call :Download %URL% %FileLocation%
echo Done

::**************************************************************************
:Download <url> <File>
Powershell.exe ^
$AllProtocols = [System.Net.SecurityProtocolType]'Ssl3,Tls,Tls11,Tls12'; ^
[System.Net.ServicePointManager]::SecurityProtocol = $AllProtocols; ^

pause
::*************************************************************************

timeout /t 30

:Disconnected
